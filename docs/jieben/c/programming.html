    一个三位数可以表示为abc，若其能满足条件a^3+b^3+c^3=abc,则它是水鲜花数 1.

    #include <stdio.h>
int main(void) {
    int g,b,s;
    for(b = 1; b <= 9; b++){
        for(s = 0; s <= 9; s++){
            for(g = 0; g <= 9; g++){
                if(b*b*b + s*s*s + g*g*g == b*100+s*10+g){
                    printf("%d\n", b*100+s*10+g);
                }
            }
        }
    }
}

void sort(int array[],int n){
    int i,j,k,t;
    for(i=0;i<n-1;i++){
        k=i;
        for(j=i+1;j<n;j++){
            if(array[j]<array[k]){
                k=j;
            }

        t=array[k];
        array[k]=array[i];
        array[i]=t;

    # 求二元一次方程的解

    #include<stdio.h>
#include<math.h>
int main(){
  int a,b,c;
  printf("请输入二元一次方程的三个系数");
  scanf("%d%d%d",&a,&b,&c);
  double delta,x1,x2;
  delta=b*b-4*a*c;
  if(delta>0){
  	x1=(-b+sqrt(delta))/(2*a);
  	x2=(-b-sqrt(delta))/(2*a);
	printf("x1=%f,x2=%f",x1,x2);
  }
  else if(delta==0){
  	x1=(-b)/(2*a);
  	x2=x1;
  	printf("x1=%f,x2=%f",x1,x2);
  }

  else{
  printf("无解!\n");
  }

  return 0;
}
